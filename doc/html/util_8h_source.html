108 template<typename InputIter>
109 std::string
110 join(InputIter lo, InputIter hi, std::string_view delim) {
111  std::string s;
112  if (lo != hi) {
113  s.insert(s.cend(), lo->begin(), lo->end());
114  while (++lo != hi) {
115  s.insert(s.cend(), delim.begin(), delim.end());
116  s.insert(s.cend(), lo->begin(), lo->end());
117  }
118  }
119  return s;
120 }
