323 template<typename F>
324 void
325 Network::consolidate(F f, Node *root) {
326  if (!root) root = &(nodes_.begin()->second);
327 
328  size_t mergers = 0;
329  do {
330  mergers = 0;
331  {
332  std::vector<Node *> children(root->children_.begin(), root->children_.end());
333  for (Node *c : children) {
334  if (f(root, c)) {
335  root->merge_child(c);
336  ++mergers;
337  }
338  }
339  }
340 
341  {
342  std::vector<Node *> siblings(root->children_.begin(), root->children_.end());
343  while (!siblings.empty()) {
344  Node *a = siblings.back();
345  siblings.pop_back();
346  for (size_t i = 0; i < siblings.size(); ) {
347  if (f(a, siblings[i])) {
348  a->merge_sibling(siblings[i]);
349  siblings[i] = siblings.back();
350  siblings.pop_back();
351  ++mergers;
352  } else {
353  ++i;
354  }
355  }
356  }
357  }
358  } while (mergers != 0);
359 
360  for (Node *c : root->children_) consolidate(f, c);
361 
362  if (root->is_root()) {
363  for (auto ii = nodes_.begin(); ii != nodes_.end(); ) {
364  if (ii->second.is_disconnected()) {
365  ii = nodes_.erase(ii);
366  } else {
367  ++ii;
368  }
369  }
370  }
371 }
